CREATE MATERIALIZED VIEW decembertemp AS select d.datetime, d.tempmin, d.tempmax from "DBWeather" d where d.datetime like '%2020-12%';
CREATE MATERIALIZED VIEW decemberhumidity AS select d.datetime, d.humidity from "DBWeather" d where d.datetime like '%-12-%';
CREATE MATERIALIZED VIEW windandtemp AS select d.datetime, d.windspeed, d.temp from "DBWeather" d where d.datetime like '%-01-%' or d.datetime like '%-02-%' or d.datetime like '%-03-%';
